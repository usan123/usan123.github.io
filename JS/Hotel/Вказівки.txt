1. У вас вже є готова робота, працюйте в index.html
2. Потібно підключити файл бібліотеки в <head>, 
після підключення css, перед </head>:
   <script src="scrollreveal.min.js"></script>
3. Створити script перед закриваючим </body> 
4. Додати туди правило:

  ScrollReveal().reveal('h1', {
        //налаштування появи, є багато варіантів
        //пробуйте по черзі додавати по одному сюди
     
        });
ScrollReveal().reveal('хто буде появлятися', {   налаштування })  - 
    'хто буде появлятися' - можна додавати теги ('p')
                            можна додавати класи ('.alala')
                            можна додавати id ('#alala')

     delay:200,// затримка перед виконанням
     duration:2000, // скільки часу виконується, в мілісекундах
     origin:'right',// left, top, bottom - напрямок, 
     distance:'1000px',//відстаньнь яку долає
     interval:500,//інтервал появи однакових елементів, тільки для меню, списку і т.д.
     reset:true,//щоб повторювалося кожен раз при пролистуванні, 
     якщо не додати - виконається один раз, після оновлення
